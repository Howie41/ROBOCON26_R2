# TailClawController 改造说明

## Controller 接口

```cpp
/**
 * @brief 尾爪控制器
 *
 * 负责尾爪机构的统一调度，包括命令读取、PC/视觉距离读取、
 * 自动对准判断、手动控制、电机输出、GPIO 输出和状态发布。
 *
 * tail_claw_task 后续只作为 FreeRTOS 任务壳，具体控制逻辑全部收敛到该类中。
 */
class TailClawController {
public:
    /**
     * @brief 获取尾爪控制器单例
     *
     * 尾爪机构只有一套硬件执行器，因此控制器使用单例，避免多个对象同时控制同一套电机和 GPIO。
     */
    static TailClawController& Instance();

    /**
     * @brief 初始化尾爪控制器
     *
     * 绑定尾爪平移电机和翻转电机，并初始化 PID、模式、目标值和匹配状态。
     *
     * @param move_motor 尾爪平移电机，使用 C610
     * @param roll_motor 尾爪翻转电机，使用 C620
     */
    void Init(C610Motor& move_motor, C620Motor& roll_motor);

    /**
     * @brief 1ms 周期总调度
     *
     * 由 tail_claw_task 周期调用。
     * 内部依次完成命令消费、距离消费、控制意图更新、目标值更新、GPIO 输出、电机输出和状态发布。
     */
    void Tick1ms();

    /**
     * @brief 处理上层下发的尾爪控制命令
     *
     * 该接口只更新命令中显式设置的字段。
     * 例如只设置翻转角度时，不会覆盖当前模式、夹爪状态或气泵状态。
     *
     * @param cmd 尾爪控制命令
     */
    void HandleCommand(const pub_tail_claw_cmd& cmd);

    /**
     * @brief 更新 PC/视觉下发的距离
     *
     * 自动对准模式下，该距离用于判断目标位置和生成平移控制目标。
     *
     * @param distance PC/视觉模块计算出的目标距离
     */
    void OnPcDistance(int16_t distance);

    /**
     * @brief 获取尾爪当前状态
     *
     * 给其他模块查询尾爪模式、目标位置、夹爪状态、气泵状态和自动匹配状态。
     *
     * @return TailClawStatus 当前尾爪状态
     */
    TailClawStatus GetStatus() const;

private:
    /**
     * @brief 私有构造函数
     *
     * 配合 Instance() 实现单例，禁止外部直接创建控制器对象。
     */
    TailClawController() = default;

    /**
     * @brief 消费尾爪控制命令 topic
     *
     * 从消息系统读取上层控制命令，并调用 HandleCommand() 更新内部状态。
     */
    void consume_commands();

    /**
     * @brief 消费 PC/视觉距离 topic
     *
     * 从消息系统读取最新距离，并调用 OnPcDistance() 缓存距离和时间戳。
     */
    void consume_pcDistance();

    /**
     * @brief 更新运动控制位
     *
     * 根据当前模式、自动对准结果或手动输入生成 motion_bits_。
     * motion_bits_ 用于表达当前周期的运动意图。
     */
    void updateControlBits();

    /**
     * @brief 更新自动对准逻辑
     *
     * 根据视觉消息和 PC 距离判断目标是否稳定匹配，并在自动模式下更新平移目标。
     *
     * @param msg 尾爪/视觉相关消息
     */
    void updateAutoAlign(const tail_claw_msg* msg);

    /**
     * @brief 更新手动控制逻辑
     *
     * 手动模式下，根据遥控器或上层命令更新目标值和执行器状态。
     */
    void updateManualControl();

    /**
     * @brief 根据运动控制位更新控制目标
     *
     * 将 motion_bits_ 转换为具体的平移目标、翻转目标或机构动作。
     */
    void updateTargetByControlBits();

    /**
     * @brief 应用 GPIO 输出
     *
     * 根据夹爪和气泵状态控制对应 GPIO。
     */
    void applyGpio();

    /**
     * @brief 应用电机输出
     *
     * 根据当前目标位置/角度计算 PID 输出，并发送给平移电机和翻转电机。
     */
    void applyMotorOutput();

    /**
     * @brief 发布尾爪状态
     *
     * 将当前模式、目标值、执行器状态和匹配状态发布给其他模块。
     */
    void publishStatus();

    /**
     * @brief 计算平移电机控制量
     *
     * 使用位置环和速度环 PID，根据目标平移位置计算 C610 输出。
     *
     * @param target_cm 目标平移位置，单位 cm
     * @return float 平移电机输出
     */
    float calcMoveCmd(float target_cm);

    /**
     * @brief 计算翻转电机控制量
     *
     * 使用位置环和速度环 PID，根据目标翻转角度计算 C620 输出。
     *
     * @param target_deg 目标翻转角度，单位 deg
     * @return float 翻转电机输出
     */
    float calcRollCmd(float target_deg);
};
```

## tail_claw_task 只作为任务壳

```cpp
void tail_claw_task(void *argument) {
    TickType_t currentTime = xTaskGetTickCount();

    TailClawController::Instance().Init(
        tail_claw_move_motor,
        tail_claw_roll_motor
    );

    for (;;) {
        TailClawController::Instance().Tick1ms();
        vTaskDelayUntil(&currentTime, 1);
    }
}
```

## 模式和命令转换

```cpp
/**
 * @brief 尾爪控制命令
 *
 * 使用 set_xxx 标志位表示对应字段是否有效。
 * 这样可以做到局部更新，避免一次命令覆盖所有尾爪状态。
 */
typedef struct {
    // 是否设置尾爪工作模式
    bool set_mode;

    // 目标工作模式，仅当 set_mode 为 true 时有效
    TailClawMode mode;

    // 是否设置翻转目标角度
    bool set_roll_target;

    // 翻转目标角度，单位 deg，仅当 set_roll_target 为 true 时有效
    float roll_target_deg;

    // 是否设置武器夹爪状态
    bool set_weapon_claw;

    // 武器夹爪是否闭合，仅当 set_weapon_claw 为 true 时有效
    bool weapon_claw_close;

    // 是否设置气泵状态
    bool set_air_pump;

    // 气泵是否开启，仅当 set_air_pump 为 true 时有效
    bool air_pump_on;

    // 是否重置自动匹配状态
    bool reset_match;
} pub_tail_claw_cmd;

void TailClawController::HandleCommand(const pub_tail_claw_cmd& cmd) {
    // 切换尾爪工作模式
    if (cmd.set_mode) {
        SetMode(cmd.mode);
    }

    // 更新翻转目标角度
    if (cmd.set_roll_target) {
        roll_target_deg_ = cmd.roll_target_deg;
    }

    // 更新武器夹爪开合状态
    if (cmd.set_weapon_claw) {
        weapon_claw_closed_ = cmd.weapon_claw_close;
    }

    // 更新气泵开关状态
    if (cmd.set_air_pump) {
        air_pump_on_ = cmd.air_pump_on;
    }

    // 重置视觉/武器匹配状态
    if (cmd.reset_match) {
        resetMatchState();
    }
}
```
